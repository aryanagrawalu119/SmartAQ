<script>

const ctx = document.getElementById('aqChart').getContext('2d');

const chart = new Chart(ctx,{

type:'line',

data:{

labels:[],

datasets:[{

label:'AQI',

data:[],

borderColor:'red',

borderWidth:2

}]

},

options:{

scales:{

y:{

beginAtZero:true

}

}

}

});

function updateDashboard(){

// Example random values (replace with API data later)

let aqi=Math.floor(Math.random()*300);

let pm=Math.floor(Math.random()*200);

let temp=(25+Math.random()*5).toFixed(1);

let hum=(50+Math.random()*10).toFixed(1);

document.getElementById("aqi").innerHTML=aqi;

document.getElementById("pm").innerHTML=pm+" µg/m³";

document.getElementById("temp").innerHTML=temp+" °C";

document.getElementById("hum").innerHTML=hum+" %";

let time=new Date().toLocaleTimeString();

chart.data.labels.push(time);

chart.data.datasets[0].data.push(aqi);

if(chart.data.labels.length>10){

chart.data.labels.shift();

chart.data.datasets[0].data.shift();

}

chart.update();

}

// update every 5 seconds

setInterval(updateDashboard,5000);

</script>
